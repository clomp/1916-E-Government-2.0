\section{Open Government}
\frame{
\frametitle{Open Government}

\begin{kasten}[Barack Obama "'Transparency and Open Government", 2009]
\begin{itemize}
 \item {\bf Government should be transparent} -- Transparenz fördert Rechenschaftspflicht und bietet Informationen für Bürger über das, was ihre Regierung tut.
 \item {\bf Government should be participatory} -- Partizipation verstärkt die Effektivität von Regierung und Verwaltung und verbessert die   Qualität ihrer Entscheidungen.
 \item {\bf Government should be collaborative}  -- Kolaboration bietet innovative Werkzeuge, Methoden und Systeme, um die Zusammenarbeit unter den Verwaltungen und mit der Öffentlichkeit zu stärken.
\end{itemize}
\end{kasten}
}

\frame{
\frametitle{Transparenz}
\begin{merkmale}[Transparenz]
	\begin{itemize}
	\item Externe Nachvollziehbarkeit von Vorgängen und Entscheidungen in Politk, Verwaltung und Justiz.
	\item Umsetzung durch soziale Netzwerke, transparente Informationssysteme und offene,gemeinschaftliche Formen des Editierens  von 
Texten und Beiträgen.
     \item Öffentlicher Zugang zu nicht-personenbezogenen und nicht geheimen Daten und Informationen der öffentlichen Verwaltung (``'Open Data''').
\end{itemize}
\end{merkmale}
}

\frame{\frametitle{Partizipation}
	
\begin{merkmale}[Partizipation]
 \begin{itemize}
  \item Einbindung der Bürger in Entscheidungsprozessen von Politk und Verwaltung.
  \item Umsetzung durch offenen gemeinschaftlichen Dialog
  \item Meinungsbildung wird durch elektronische Medien Text, Bild, Ton und Video ergänzt.
  \item ergebnisoffene Bürgerbefragungen
  \item Bewertungen und Meinungsbildgewinnung auf Knopfdruck
  \item moderierter Dialog, verteiltes Brainstorming
\end{itemize}
\end{merkmale}
}

\frame{\frametitle{Kollaboration}
 \begin{merkmale}[Kollaboration]
  \begin{itemize}
    \item Verstärkte Einbindung von Bürgern, Unternehmen, Verbände und Bürgerinitiativen in die Aufgabenverteilung
zwischen Staat und Gesellschaft.
    \item Verstärkte übergreifende und interne Zusammenarbeit von Behörden (shared services)
    \item Einbindung von Bürgern, Verbänden und Unternehmen in vorhandene Prozessketten.
    \item Schwarmauslagerung (``'Crowd Sourcing'''), z.B. durch Open Source, Government Mashups, AppStores, Hackdays
   \end{itemize}
 \end{merkmale}

}

\frame{\frametitle{Open Innovation}
 \begin{merkmale}[Öffnung des Innovationsprozesses]
  \begin{itemize}
    \item Innovation der Verwaltung durch Ideen der Bürger, Programmierer und Unternehmen.
    \item Protypen durch Datenportale oder Ideen-Wettbewerbe (``'Apps4Government''')
    \item Innovationsprozess als Wirtschaftsförderungsmaßnahme.
    \item Personalpolitik
   \end{itemize}
 \end{merkmale}
}


\frame{
\frametitle{Öffnung gegenüber der Gesellschaft}
 \begin{merkmale}[Öffnung gegenüber der Gesellschaft]
  \begin{itemize}
    \item Ziele: aktiver Austausch und ergebnisoffener Dioalog und Diskurs von Bürgern und Verwaltung
    \item Ausrichtung der Verwaltung auf Bedürfnisse und Probleme der Bürger führt zu erhöhter Bürgerzufriedenheit
    \item partnerschaftliches Verhältnis, gegenseitiges Vertrauen.
   \end{itemize}
 \end{merkmale}
}

\frame{
 \frametitle{Open Data}
 \begin{merkmale}[Open Data]
  \begin{itemize}
  \item Behördendaten und mit öffentlichen Mitteln finanzierte Daten öffentlich zugänglich machen.
  \item Vollständige  Primärquellen, zeitnah nach ihrer Generierung.
  \item Keine Diskriminierungen und  Einschränkungen beim Zugriff.
  \item Vernetzung der Daten
  \item Generiern neuen Wissens (``'HackDays''')
  \item Freie unlizensierte Inhalte wie Texte, Bild-,Ton- und Filmwerke (z.B. Dt. Digitale Bibiothek oder Digt. Bildarchiv des BAarch)
   \end{itemize}
 \end{merkmale}
}

\frame{
 \frametitle{Offene Standards, Offene Schnittstellen}
 \begin{merkmale}
  \begin{itemize}
  \item Grundlage für offene Kommunikationssysteme  
  \item Interoperabilität durch Offenheit, z.B. Datenformat ODF
  \item Europäischer Interoperabilitätsrahmenwerk 
  \end{itemize}
 \end{merkmale}
}

\section{E-Government 2.0}
\frame{\frametitle{Web 2.0 Technologie}
\begin{itemize}
\item Blogs,  Wikis  und  offene  Redaktionssysteme  ermöglichen  ein gemeinsames, verteiltes Editieren von Texten. 
\item Foren und Beratungsdienste eröffnen Formen des gemeinschaftlichen Diskutierens. 
\item Gemeinschaftlichen Entscheidungsfindung und  verteiltes  Programmieren.
\item Transparente, partizipative und kollaborative Ansätze  zur Verwaltungsmodernisierung durch Web 2.0 
\end{itemize}
}
